# ######################################################################################################################
# Target name
# ######################################################################################################################

set(TARGET_NAME pbconvert)

# ######################################################################################################################
# Sources
# ######################################################################################################################

set(SOURCES common.cpp)

if(WITH_IAM_API)
    list(APPEND SOURCES iam.cpp)
endif()

if(WITH_SM_API)
    list(APPEND SOURCES sm.cpp)
endif()

# ######################################################################################################################
# Compile options
# ######################################################################################################################

set(COMPILE_OPTIONS -Wno-format-truncation)

# ######################################################################################################################
# Includes
# ######################################################################################################################

set(INCLUDES ${AOS_COMMON_DIR})

# ######################################################################################################################
# Libraries
# ######################################################################################################################

set(LIBRARIES ${TARGET_NAMESPACE}::utils aoscommon Poco::Util)

if(WITH_IAM_API)
    list(APPEND LIBRARIES aoscoreapi-gen-iam)
endif()

if(WITH_SM_API)
    list(APPEND LIBRARIES aoscoreapi-gen-sm)
endif()

# ######################################################################################################################
# Target
# ######################################################################################################################

add_module(
    TARGET_NAME
    ${TARGET_NAME}
    LOG_MODULE
    STACK_USAGE
    ${AOS_STACK_USAGE}
    SOURCES
    ${SOURCES}
    COMPILE_OPTIONS
    ${COMPILE_OPTIONS}
    INCLUDES
    ${INCLUDES}
    LIBRARIES
    ${LIBRARIES}
)
