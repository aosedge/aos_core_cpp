# ######################################################################################################################
# Target name
# ######################################################################################################################

set(TARGET_NAME boot)

# ######################################################################################################################
# Sources
# ######################################################################################################################

set(SOURCES boot.cpp config.cpp eficontroller.cpp partitionmanager.cpp)

# ######################################################################################################################
# Libraries
# ######################################################################################################################

find_package(PkgConfig REQUIRED)

pkg_check_modules(EFIVAR efivar REQUIRED)
pkg_check_modules(EFIBOOT efiboot REQUIRED)
pkg_check_modules(BLKID blkid REQUIRED)

set(LIBRARIES
    aos::core::common::tools
    aos::sm::config
    aos::sm::utils
    aos::sm::runtimes::utils
    ${EFIVAR_LIBRARIES}
    ${EFIBOOT_LIBRARIES}
    ${BLKID_LIBRARIES}
)

# ######################################################################################################################
# Target
# ######################################################################################################################

add_module(
    TARGET_NAME
    ${TARGET_NAME}
    LOG_MODULE
    STACK_USAGE
    ${AOS_STACK_USAGE}
    SOURCES
    ${SOURCES}
    HEADERS
    ${HEADERS}
    LIBRARIES
    ${LIBRARIES}
)

# ######################################################################################################################
# Tests
# ######################################################################################################################

if(WITH_TEST)
    add_subdirectory(tests)
endif()
